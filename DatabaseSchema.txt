CREATE TABLE USERS
(
    USERNAME VARCHAR(60) NOT NULL PRIMARY KEY,
    PASSWORD VARCHAR(60) NOT NULL
);

CREATE TABLE WAIFUS
(
    USERNAME VARCHAR(60),
    NAME VARCHAR(60) NOT NULL PRIMARY KEY,
    HAIR INT,
    FACE INT,
    OUTFIT INT,
    EYES INT,
    HAT INT,
    FACACC INT,
    UBODY INT,
    LBODY INT,
    GOODT1 INT,
    GOODT2 INT,
    GOODT3 INT,
    BADT INT
);
